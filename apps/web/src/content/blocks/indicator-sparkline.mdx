## Overview

`IndicatorSparkline` renders a compact sparkline line chart suitable for inline display
alongside each indicator row in the CBBI breakdown table. It shows the last N days of
history for a single sub-score, with the current value highlighted as an endpoint dot.

## Usage

```tsx filename="IndicatorRow.tsx"
import { IndicatorSparkline } from '@/components/charts/IndicatorSparkline'

interface IndicatorRowProps {
  name: string
  currentScore: number
  history: number[]  // last 90 days, newest last
}

export function IndicatorRow({ name, currentScore, history }: IndicatorRowProps) {
  return (
    <tr>
      <td>{name}</td>
      <td>
        <IndicatorSparkline data={history} current={currentScore} />
      </td>
      <td>{currentScore}</td>
    </tr>
  )
}
```

## Props

| Prop | Type | Required | Default | Description |
|-|-|-|-|-|
| `data` | `number[]` | Yes | — | Historical sub-scores (0–100), oldest first |
| `current` | `number` | Yes | — | Current score value — renders as endpoint dot |
| `width` | `number` | No | `120` | Pixel width |
| `height` | `number` | No | `32` | Pixel height |
| `color` | `string` | No | auto | Line color — defaults to intent color based on `current` |
| `showDot` | `boolean` | No | `true` | Render the endpoint dot |

## Color mapping

When `color` is not provided, the line color maps to Blueprint intent based on the current value:

```typescript
function scoreToColor(score: number): string {
  if (score >= 80) return 'var(--bp-intent-danger)'
  if (score >= 60) return 'var(--bp-intent-warning)'
  if (score >= 40) return 'var(--bp-intent-primary)'
  return 'var(--bp-intent-success)'
}
```

This matches the color scale used in the CBBI gauge chart and the indicator heatmap.

## Implementation

```tsx filename="IndicatorSparkline.tsx"
import { LineChart, Line, ResponsiveContainer, ReferenceDot } from 'recharts'

interface Props {
  data: number[]
  current: number
  width?: number
  height?: number
  color?: string
  showDot?: boolean
}

export function IndicatorSparkline({ data, current, width = 120, height = 32, color, showDot = true }: Props) {
  const resolvedColor = color ?? scoreToColor(current)
  const chartData = data.map((value, i) => ({ value, i }))

  return (
    <LineChart width={width} height={height} data={chartData}>
      <Line
        type="monotone"
        dataKey="value"
        stroke={resolvedColor}
        strokeWidth={1.5}
        dot={false}
        isAnimationActive={false}
      />
      {showDot && (
        <ReferenceDot
          x={chartData.length - 1}
          y={current}
          r={3}
          fill={resolvedColor}
          stroke="none"
        />
      )}
    </LineChart>
  )
}
```

<Admonition type="tip" title="isAnimationActive={false}">
  Disable Recharts animation for sparklines rendered inside tables or lists. Animating dozens of
  sparklines simultaneously causes visible jank and increases layout recalculation cost.
</Admonition>

## Dark mode

The component uses Blueprint CSS custom properties for all colors. No additional dark mode
handling is needed — the correct token value resolves automatically when the `bp5-dark` class
is active on `<body>`.
